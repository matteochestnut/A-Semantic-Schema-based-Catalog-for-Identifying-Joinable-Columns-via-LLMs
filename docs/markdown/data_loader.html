 46    def _truncate_text(self,
 47        text: str,
 48        max_tokens: int,
 49        model_name: str = 'text-embedding-ada-002'
 50    ) -> str:
 51        """
 52        Private helper method to truncate text to fit within the token limit for the embedding model.
 53
 54        Args:
 55            text (str): The text to truncate.
 56            max_tokens (int): The maximum number of tokens allowed for the embedding model.
 57            model_name (str): The name of the embedding model to use.
 58
 59        Returns:
 60            str: The truncated text, or the original text if it is already within the token limit.
 61        """
 62        if self.tokenizer:
 63            encoded_text = self.tokenizer.encode(text)
 64        else:
 65            try:
 66                encoding = tiktoken.encoding_for_model(model_name)
 67            except KeyError:
 68                encoding = tiktoken.get_encoding("cl100k_base")
 69            encoded_text = encoding.encode(text)
 70
 71        if len(encoded_text) > max_tokens:
 72            truncated_text = self.tokenizer.decode(encoded_text[:max_tokens]) if self.tokenizer else encoding.decode(encoded_text[:max_tokens])
 73            print(add_timestamp(color_text(f"Truncated text from {len(encoded_text)} to {max_tokens} tokens.", "yellow")))
 74            return truncated_text
 75        return text

 77    def _format_date(self,
 78        date_str: str
 79    ) -> str | None:
 80        """
 81        Private helper method to convert date strings to standardized format.
 82
 83        This method takes a string date_str and attempts to convert it to
 84        a standardized format of 'YYYY-MM-DDTHH:MM' using various date formats.
 85        If the conversion is successful, it returns the converted date string.
 86        Otherwise, it returns None.
 87
 88        This method is used internally by the DataLoader to convert date columns
 89        to a standardized format. The supported date formats are:
 90            - YYYY-MM-DDTHH:MM (ISO 8601 format)
 91            - YYYY-MM-DD (date only)
 92            - YYYY-MM (date only, with month as a number)
 93            - YYYY-QX (quarterly dates, where X is 1-4)
 94            - YYYY-SX or YYYY-HX (semester dates, where X is 1 or 2)
 95            - YYYYMMDD (date only, with month and day as numbers)
 96            - MM/YYYY or MM-YYYY (date only, with month as a number)
 97            - Jan 2023, January 2023, etc. (textual dates)
 98
 99        If the input date string does not match any of the supported formats,
100        it returns None.
101        """
102        if not date_str or pd.isna(date_str):
103            return None
104
105        date_str = str(date_str).strip()
106
107        try:
108            # Handle quarterly dates (e.g., 2024-Q3)
109            if 'Q' in date_str and len(date_str.split('-Q')) == 2 and date_str.split('-Q')[0].isdigit() and date_str.split('-Q')[1].isdigit():
110                year, quarter = map(int, date_str.split('-Q'))
111                month = (quarter - 1) * 3 + 1
112                dt = datetime(year, month, 1)
113                return dt.strftime('%Y-%m-%dT%H:%M')
114
115            # Handle YYYY-MM (e.g., 2005-03)
116            if len(date_str) == 7 and date_str[4] == '-':
117                year, month = map(int, date_str.split('-'))
118                if 1 <= month <= 12:
119                    dt = datetime(year, month, 1)
120                    return dt.strftime('%Y-%m-%dT%H:%M')
121
122            # Handle semester dates (e.g., 2023-S1, 2023-H1)
123            if any(s in date_str for s in ['-S1', '-S2', '-H1', '-H2']):
124                parts = re.split(r'-[SH]', date_str)
125                if len(parts) == 2 and parts[0].isdigit() and parts[1].isdigit():
126                    year, semester = int(parts[0]), int(parts[1])
127                    month = 1 if semester == 1 else 7
128                    dt = datetime(year, month, 1)
129                    return dt.strftime('%Y-%m-%dT%H:%M')
130
131            # Handle YYYYMMDD (e.g., 20230101)
132            if len(date_str) == 8 and date_str.isdigit():
133                year, month, day = int(date_str[:4]), int(date_str[4:6]), int(date_str[6:8])
134                if 1 <= month <= 12 and 1 <= day <= 31:
135                    dt = datetime(year, month, day)
136                    return dt.strftime('%Y-%m-%dT%H:%M')
137
138            # Handle MM/YYYY or MM-YYYY
139            if isinstance(date_str, str) and len(date_str) == 7 and (date_str[2] == '/' or date_str[2] == '-'):
140                try:
141                    if date_str[2] == '/':
142                        month_str, year_str = date_str.split('/')
143                    else:
144                        month_str, year_str = date_str.split('-')
145
146                    if month_str.isdigit() and year_str.isdigit():
147                        month = int(month_str)
148                        year = int(year_str)
149                        if 1 <= month <= 12:
150                            dt = datetime(year, month, 1)
151                            return dt.strftime('%Y-%m-%dT%H:%M')
152                except ValueError:
153                    pass
154
155            # Handle textual dates (e.g., "Jan 2023", "January 2023")
156            if isinstance(date_str, str) and any(month in date_str for month in ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']):
157                try:
158                    for fmt in ['%b %Y', '%B %Y', '%b, %Y', '%B, %Y']:
159                        try:
160                            dt = datetime.strptime(date_str, fmt)
161                            return dt.strftime('%Y-%m-%dT%H:%M')
162                        except ValueError:
163                            continue
164                except Exception:
165                    pass
166
167            # Try different date formats
168            for fmt in [
169                '%Y-%m-%d', '%d/%m/%Y', '%Y/%m/%d', '%d-%m-%Y', '%Y',
170                '%d.%m.%Y', '%m/%d/%Y', '%Y.%m.%d', '%d %b %Y', '%d %B %Y',
171                '%b %d, %Y', '%B %d, %Y', '%Y-%m-%dT%H:%M:%S', '%Y-%m-%dT%H:%M',
172                '%d/%m/%Y %H:%M', '%d/%m/%Y %H:%M:%S'
173            ]:
174                try:
175                    dt = datetime.strptime(date_str, fmt)
176                    return dt.strftime('%Y-%m-%dT%H:%M')
177                except ValueError:
178                    continue
179
180            return None # If none of the formats match
181        except Exception as e:
182            print(add_timestamp(color_text(f"Error formatting date '{date_str}': {e}", "red")))
183            return None

185    def _get_distinct_sample_values(self,
186        df: pd.DataFrame,
187        column: str,
188        n: int = 100
189    ) -> list[str]:
190        """
191        Retrieves up to 'n' distinct non-null values from the specified column in a DataFrame.
192
193        This method extracts unique values from the given column, excluding any null values.
194        If the number of distinct values exceeds 'n', a random sample of 'n' values is returned.
195        Otherwise, all distinct values are returned.
196
197        Args:
198            df (pd.DataFrame): The DataFrame from which to extract distinct values.
199            column (str): The column name to process within the DataFrame.
200            n (int, optional): The maximum number of distinct values to return. Defaults to 100.
201
202        Returns:
203            list[str]: A list of up to 'n' distinct non-null values from the specified column.
204        """
205        distinct_values = df[column].dropna().unique()
206        if len(distinct_values) > n:
207            return random.sample(list(distinct_values), n)
208        return list(distinct_values)

210    def _create_llm_description_prompt(self,
211        file_info: dict
212    ) -> str:
213        """
214        Creates a structured prompt for the LLM to analyze and describe a dataset.
215
216        This method constructs a textual prompt that provides various details about a dataset,
217        including file information, temporal coverage, headers, column samples, and metadata 
218        descriptions if available. The prompt is used to instruct the LLM to generate a 
219        comprehensive analysis of the dataset.
220
221        Args:
222            file_info (dict): A dictionary containing information about the dataset file, 
223                            including file name, title, headers, column samples, metadata 
224                            descriptions, temporal start and end dates, and coverage sector.
225
226        Returns:
227            str: A structured prompt string to be used by the LLM for dataset analysis.
228        """
229
230        file_name = file_info['file_name']
231        title = file_info.get('title', '')
232        headers = file_info['original_headers']
233        sample_data = {col['column_name']: col['values_sample'] for col in file_info['columns']}
234        json_column_descriptions = file_info.get('metadata_description', False)
235        start_date = file_info.get('temporal_start_date', None)
236        end_date = file_info.get('temporal_end_date', None)
237        coverage_sector = file_info.get('coverage_sector', '')
238
239        prompt = f"""Analyze this dataset and provide a structured analysis.
240
241FILE INFORMATION:
242- File name: {file_name}
243- Title: {title}
244"""
245
246        if start_date or end_date:
247            prompt += "\nTEMPORAL INFORMATION:\n"
248            if start_date:
249                prompt += f"- Start date: {start_date}\n"
250            if end_date:
251                prompt += f"- End date: {end_date}\n"
252
253        if coverage_sector:
254            prompt += f"\nCOVERAGE SECTOR:\n{coverage_sector}\n"
255
256        prompt += f"""
257HEADERS:
258{', '.join(headers)}
259
260COLUMN SAMPLES (up to 10 distinct values per column):
261"""
262        for column in headers:
263            prompt += f"\n{column}: {sample_data.get(column, [])}"
264            
265
266        if json_column_descriptions:
267            prompt += "\n\nCOLUMN DESCRIPTIONS (from metadata, if available):"
268            for column in file_info["columns"]:
269                description = column["metadata_description"]
270                header = column["header"]
271                prompt += f"\n{header}: {description}"
272
273        prompt += """
274
275TASK:
276Based on the data provided, generate a structured analysis with the following components:
277
278RESPONSE FORMAT:
279{
280    "description": "A comprehensive description of what this dataset contains and represents",
281    "column_descriptions": {
282        "column1": "description of column1",
283        "column2": "description of column2",
284        ...
285    },
286    "spatial_coverage": "The geographical areas covered by this dataset",
287    "reasoning": "Explanation of your analysis and how you determined the descriptions and coverage"
288}
289
290RULES:
291- Provide clear, concise descriptions for each field.
292- For column descriptions, include all columns from the headers, even if no metadata description was provided.
293- Base your spatial coverage on country related columns if present, try to generalize an area.
294- Use the coverage sector information to enhance your understanding of the dataset.
295- In the reasoning section, explain your thought process and any patterns you observed.
296"""
297        return prompt

299    def _get_llm_description(self,
300        prompt: str
301    ) -> dict:
302        """
303        Obtains a structured analysis of a dataset using a Large Language Model (LLM).
304
305        Args:
306            prompt (str): The structured prompt to use for the LLM analysis.
307
308        Returns:
309            dict: A dictionary with the following keys: description, column_descriptions, spatial_coverage, reasoning.
310        """
311        if not self.openai_client:
312            print(add_timestamp(color_text("OpenAI client not initialized. Skipping LLM description analysis.", "red")))
313            return {
314                "description": "LLM skipped: OpenAI client not initialized.",
315                "column_descriptions": {},
316                "spatial_coverage": "LLM skipped: OpenAI client not initialized.",
317                "reasoning": "LLM skipped: OpenAI client not initialized."
318            }
319        try:
320            response = self.openai_client.chat.completions.create(
321                model="gpt-4o", # Can be made configurable
322                messages=[
323                    {"role": "system", "content": "You are a data analyst specializing in statistical datasets. You provide structured, accurate analyses of data. Always respond in the exact JSON format requested."},
324                    {"role": "user", "content": prompt}
325                ],
326                temperature=0.1,
327                response_format={"type": "json_object"}
328            )
329            res = json.loads(response.choices[0].message.content)
330            return res
331        except Exception as e:
332            print(add_timestamp(color_text(f"Error getting LLM analysis: {e}", "red")))
333            return {
334                "description": f"Error in LLM analysis: {e}",
335                "column_descriptions": {},
336                "spatial_coverage": "Unknown",
337                "reasoning": f"Error: {e}"
338            }

340    def _get_unified_description(self,
341        column: str,
342        descriptions: list[str]
343    ) -> str:
344        """
345        Unify multiple descriptions for a single column into a single comprehensive description.
346
347        Args:
348            column (str): The column name for the descriptions to be unified.
349            descriptions (list[str]): A list of descriptions for the column.
350
351        Returns:
352            str: A single unified description for the column. If an error occurs, returns a semicolon joined list of the input descriptions.
353        """
354        try:
355            prompt = f"""Unify the following descriptions for the column "{column}" into a single comprehensive description:
356
357DESCRIPTIONS:
358"""
359            for i, desc in enumerate(descriptions, 1):
360                prompt += f"{i}. {desc}\n"
361
362            prompt += """
363TASK:
364Create a single unified description that captures the essence of all these descriptions.
365The unified description should be clear, comprehensive, and concise.
366
367RESPONSE FORMAT:
368{
369    "unified_description": "Your unified description here"
370}
371"""
372            response = self.openai_client.chat.completions.create(
373                model="gpt-4o", # Can be made configurable
374                messages=[
375                    {"role": "system", "content": "You are a data modeling expert specializing in creating unified descriptions for data columns. Always respond in JSON format."},
376                    {"role": "user", "content": prompt}
377                ],
378                temperature=0.1,
379                response_format={"type": "json_object"}
380            )
381            result = json.loads(response.choices[0].message.content)
382            return result.get("unified_description", "; ".join(descriptions)) # Fallback
383        except Exception as e:
384            print(add_timestamp(color_text(f"Error unifying descriptions for {column}: {e}", "red")))
385            return "; ".join(descriptions) # Fallback to joining descriptions

387    def _unify_column_data(self,
388        data_lake_list: list[dict]
389    ) -> list[dict]:
390        """
391        Unifies column descriptions in the provided data lake list using an LLM.
392
393        This function consolidates multiple descriptions for each column across the data lake
394        into a single unified description using a Large Language Model (LLM). It updates
395        each column's description with the unified result and returns the modified data lake list.
396
397        Args:
398            data_lake_list (list[dict]): A list of dictionaries representing files in the data lake,
399                                        each containing column data with descriptions.
400
401        Returns:
402            list[dict]: The updated data lake list with unified column descriptions.
403        """
404        print(add_timestamp(color_text("Unifying column data...", "yellow")))
405        all_columns = defaultdict(list)
406
407        # Collect all column data
408        for file_dict in data_lake_list:
409            for column_dict in file_dict['columns']:
410                column_header = column_dict['header']
411                if column_dict.get('llm_description'):
412                    all_columns[column_header].append(column_dict['llm_description'])
413
414        unified_columns = {}
415        for column_header, descriptions in tqdm(all_columns.items(), desc="Unifying descriptions with LLM"):
416            if len(set(descriptions)) > 1:
417                # Use LLM to unify descriptions if there are multiple unique ones
418                unified_desc = self._get_unified_description(column_header, list(set(descriptions)))
419            elif descriptions:
420                unified_desc = descriptions[0]
421            else:
422                unified_desc = "N/A"
423
424            unified_columns[column_header] = unified_desc
425        
426        # Update the original data_lake_list with the unified data
427        for file_dict in data_lake_list:
428            for column_dict in file_dict['columns']:
429                column_header = column_dict['header']
430                unified_data = unified_columns.get(column_header)
431                if unified_data:
432                    column_dict['llm_description'] = unified_data
433
434        print(add_timestamp(color_text("Column data unification complete.", "green")))
435        return data_lake_list

437    def load_and_describe_datalake(self,
438        data_directory: str,
439        metadata_directory: str | None = None,
440        llm: bool = True,
441        sample_size: int = 100,
442        max_prompt_tokens: int = 8000,
443        output_directory: str | None = None
444    ) -> list[dict]:
445        """
446        Loads data from CSV/TSV files in the provided data directory and creates a
447        data lake list with descriptions generated using a Large Language Model (LLM).
448        
449        Args:
450            data_directory (str): Path to the directory containing CSV/TSV files.
451            metadata_directory (str | None): Path to the directory containing JSON metadata
452                                             files matching the data files (optional).
453            llm (bool): Whether to use an LLM to generate descriptions for files and columns.
454            sample_size (int): Number of distinct values to sample from each column for statistics.
455            max_prompt_tokens (int): Maximum number of tokens to use in the LLM prompt.
456            output_directory (str | None): Directory to save the generated data lake JSON file to.
457        
458        Returns:
459            list[dict]: A list of dictionaries, each representing a data file with its columns,
460                        containing statistical information and descriptions generated by the LLM.
461        """
462        print(add_timestamp(color_text(f"Starting data lake loading and description from {data_directory}", "magenta")))
463        data_lake_list = []
464
465        csv_files = [f for f in os.listdir(data_directory) if f.endswith(('.csv', '.tsv'))]
466        if not csv_files:
467            print(add_timestamp(color_text(f"No CSV/TSV files found in {data_directory}.", "yellow")))
468            return []
469
470        for file_name in tqdm(csv_files, desc="Processing data files"):
471            file_path = os.path.join(data_directory, file_name)
472            sep = '\t' if file_name.endswith(".tsv") else ','
473
474            try:
475                df = pd.read_csv(file_path, sep=sep, low_memory=False, dtype=str)
476                if df.empty:
477                    print(add_timestamp(color_text(f"Skipping empty file: {file_name}", "yellow")))
478                    continue
479
480                file_info = {
481                    "file_name": file_name,
482                    "file_path": file_path, # Store full path for later access if needed
483                    "title": "",
484                    "llm_description": "",
485                    "spatial_coverage_llm": None,
486                    "temporal_start_date": None,
487                    "temporal_end_date": None,
488                    "data_themes": None,
489                    "coverage_sector": None,
490                    "original_headers": list(df.columns),
491                    "columns": [],
492                    "metadata_description": False
493                }
494                
495                metadata_column_descriptions = {}
496
497                # --- Integrate Metadata (if provided) ---
498                json_metadata = {}
499                if metadata_directory:
500                    # Try common naming conventions for metadata files
501                    base_name_no_ext = os.path.splitext(file_name)[0]
502                    metadata_candidates = [
503                        f"{base_name_no_ext}.json",
504                        f"{base_name_no_ext.replace('_data', '_metadata')}.json", # e.g., customers_data.csv -> customers_metadata.json
505                    ]
506                    found_metadata_path = None
507                    for candidate in metadata_candidates:
508                        potential_path = os.path.join(metadata_directory, candidate)
509                        if os.path.exists(potential_path):
510                            found_metadata_path = potential_path
511                            break
512
513                    if found_metadata_path:
514                        try:
515                            with open(found_metadata_path, 'r', encoding='utf-8') as f:
516                                json_metadata = json.load(f)
517                            print(add_timestamp(color_text(f"Found metadata for {file_name}: {os.path.basename(found_metadata_path)}", "blue")))
518
519                            file_info["title"] = json_metadata.get('title', '')
520                            file_info["temporal_start_date"] = self._format_date(json_metadata.get('startDate', ''))
521                            file_info["temporal_end_date"] = self._format_date(json_metadata.get('endDate', ''))
522                            file_info["data_themes"] = json_metadata.get('dataThemes', json_metadata.get('General', ''))
523                            file_info["coverage_sector"] = json_metadata.get('coverageSector', '')
524                            file_info["metadata_description"] = True
525
526                            # Extract column descriptions from JSON metadata
527                            for col_name in df.columns:
528                                if col_name in json_metadata and json_metadata[col_name] and json_metadata[col_name] != " - ":
529                                    metadata_column_descriptions[col_name] = json_metadata[col_name]
530
531                        except Exception as e:
532                            print(add_timestamp(color_text(f"Error loading/parsing metadata for {file_name} from {found_metadata_path}: {e}", "red")))
533                    else:
534                        print(add_timestamp(color_text(f"No matching metadata JSON found for {file_name} in {metadata_directory}", "yellow")))
535
536                # --- Process Columns for Statistics and Samples ---
537                for col_name in df.columns:
538                    values = df[col_name].dropna().astype(str).tolist()
539                    num_values = len(values)
540
541                    # Basic statistics
542                    if num_values > 0:
543                        lengths = [len(v) for v in values]
544                        max_length = max(lengths)
545                        min_length = min(lengths)
546                        avg_length = sum(lengths) / num_values
547                        counter = Counter(values)
548                        most_common_values = [val for val, _ in counter.most_common(min(20, num_values))] # Limit to 20
549                    else:
550                        max_length, min_length, avg_length = 0, 0, 0
551                        most_common_values = []
552
553                    column_dict = {
554                        "column_name": col_name,
555                        "header": col_name, # Original header
556                        "values_sample": self._get_distinct_sample_values(df, col_name, sample_size),
557                        "value_stats": {
558                            "num_values": num_values,
559                            "max_length": max_length,
560                            "min_length": min_length,
561                            "avg_length": round(avg_length, 2),
562                            "most_common_values": most_common_values
563                        },
564                        "metadata_description": metadata_column_descriptions.get(col_name) or None,
565                        "llm_description": None, # To be filled by LLM
566                        "semantic_annotation": None, # To be filled by SemanticAnnotator later (label)
567                    }
568                    file_info["columns"].append(column_dict)
569
570                if llm:
571                    # --- Generate LLM Descriptions for File and Columns ---
572                    llm_prompt = self._create_llm_description_prompt(file_info)
573                    llm_prompt_truncated = self._truncate_text(llm_prompt, max_prompt_tokens)
574                    llm_response = self._get_llm_description(llm_prompt_truncated)
575
576                    file_info["llm_description"] = llm_response.get("description", "")
577                    file_info["llm_spatial_coverage"] = llm_response.get("spatial_coverage", "")
578
579                    for col_dict in file_info["columns"]:
580                        col_name = col_dict["column_name"]
581                        col_dict["llm_description"] = llm_response["column_descriptions"].get(col_name, "")
582                
583                data_lake_list.append(file_info)
584                
585            except Exception as e:
586                print(add_timestamp(color_text(f"Error processing file {file_name}: {e}", "red")))
587                continue
588            
589        if llm:
590            data_lake_list = self._unify_column_data(data_lake_list)
591            
592        # --- Save Data Lake JSON ---
593        if output_directory:
594            try:

 48        text: str,
 49        max_tokens: int,
 50        model_name: str = 'text-embedding-ada-002'

 52        """

 78    def _format_date(self,
 79        date_str: str
 80    ) -> str | None:
 81        """
 82        Private helper method to convert date strings to standardized format.
 83
 84        This method takes a string date_str and attempts to convert it to
 85        a standardized format of 'YYYY-MM-DDTHH:MM' using various date formats.
 86        If the conversion is successful, it returns the converted date string.
 87        Otherwise, it returns None.
 88
 89        This method is used internally by the DataLoader to convert date columns
 90        to a standardized format. The supported date formats are:
 91            - YYYY-MM-DDTHH:MM (ISO 8601 format)
 92            - YYYY-MM-DD (date only)
 93            - YYYY-MM (date only, with month as a number)
 94            - YYYY-QX (quarterly dates, where X is 1-4)
 95            - YYYY-SX or YYYY-HX (semester dates, where X is 1 or 2)
 96            - YYYYMMDD (date only, with month and day as numbers)
 97            - MM/YYYY or MM-YYYY (date only, with month as a number)
 98            - Jan 2023, January 2023, etc. (textual dates)
 99
100        If the input date string does not match any of the supported formats,
101        it returns None.
102        """
103        if not date_str or pd.isna(date_str):
104            return None
105
106        date_str = str(date_str).strip()
107
108        try:
109            # Handle quarterly dates (e.g., 2024-Q3)
110            if 'Q' in date_str and len(date_str.split('-Q')) == 2 and date_str.split('-Q')[0].isdigit() and date_str.split('-Q')[1].isdigit():
111                year, quarter = map(int, date_str.split('-Q'))
112                month = (quarter - 1) * 3 + 1
113                dt = datetime(year, month, 1)
114                return dt.strftime('%Y-%m-%dT%H:%M')
115
116            # Handle YYYY-MM (e.g., 2005-03)
117            if len(date_str) == 7 and date_str[4] == '-':
118                year, month = map(int, date_str.split('-'))
119                if 1 <= month <= 12:
120                    dt = datetime(year, month, 1)
121                    return dt.strftime('%Y-%m-%dT%H:%M')
122
123            # Handle semester dates (e.g., 2023-S1, 2023-H1)
124            if any(s in date_str for s in ['-S1', '-S2', '-H1', '-H2']):
125                parts = re.split(r'-[SH]', date_str)
126                if len(parts) == 2 and parts[0].isdigit() and parts[1].isdigit():
127                    year, semester = int(parts[0]), int(parts[1])
128                    month = 1 if semester == 1 else 7
129                    dt = datetime(year, month, 1)
130                    return dt.strftime('%Y-%m-%dT%H:%M')
131
132            # Handle YYYYMMDD (e.g., 20230101)
133            if len(date_str) == 8 and date_str.isdigit():
134                year, month, day = int(date_str[:4]), int(date_str[4:6]), int(date_str[6:8])
135                if 1 <= month <= 12 and 1 <= day <= 31:
136                    dt = datetime(year, month, day)
137                    return dt.strftime('%Y-%m-%dT%H:%M')
138
139            # Handle MM/YYYY or MM-YYYY
140            if isinstance(date_str, str) and len(date_str) == 7 and (date_str[2] == '/' or date_str[2] == '-'):
141                try:
142                    if date_str[2] == '/':
143                        month_str, year_str = date_str.split('/')
144                    else:
145                        month_str, year_str = date_str.split('-')
146
147                    if month_str.isdigit() and year_str.isdigit():
148                        month = int(month_str)
149                        year = int(year_str)
150                        if 1 <= month <= 12:
151                            dt = datetime(year, month, 1)
152                            return dt.strftime('%Y-%m-%dT%H:%M')
153                except ValueError:
154                    pass
155
156            # Handle textual dates (e.g., "Jan 2023", "January 2023")
157            if isinstance(date_str, str) and any(month in date_str for month in ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']):
158                try:
159                    for fmt in ['%b %Y', '%B %Y', '%b, %Y', '%B, %Y']:
160                        try:
161                            dt = datetime.strptime(date_str, fmt)
162                            return dt.strftime('%Y-%m-%dT%H:%M')
163                        except ValueError:
164                            continue
165                except Exception:
166                    pass
167
168            # Try different date formats
169            for fmt in [
170                '%Y-%m-%d', '%d/%m/%Y', '%Y/%m/%d', '%d-%m-%Y', '%Y',
171                '%d.%m.%Y', '%m/%d/%Y', '%Y.%m.%d', '%d %b %Y', '%d %B %Y',
172                '%b %d, %Y', '%B %d, %Y', '%Y-%m-%dT%H:%M:%S', '%Y-%m-%dT%H:%M',
173                '%d/%m/%Y %H:%M', '%d/%m/%Y %H:%M:%S'
174            ]:
175                try:
176                    dt = datetime.strptime(date_str, fmt)
177                    return dt.strftime('%Y-%m-%dT%H:%M')
178                except ValueError:
179                    continue
180
181            return None # If none of the formats match
182        except Exception as e:
183            print(add_timestamp(color_text(f"Error formatting date '{date_str}': {e}", "red")))
184            return None

186    def _get_distinct_sample_values(self,
187        df: pd.DataFrame,
188        column: str,
189        n: int = 100
190    ) -> list[str]:
191        """
192        Retrieves up to 'n' distinct non-null values from the specified column in a DataFrame.
193
194        This method extracts unique values from the given column, excluding any null values.
195        If the number of distinct values exceeds 'n', a random sample of 'n' values is returned.
196        Otherwise, all distinct values are returned.
197
198        Args:
199            df (pd.DataFrame): The DataFrame from which to extract distinct values.
200            column (str): The column name to process within the DataFrame.
201            n (int, optional): The maximum number of distinct values to return. Defaults to 100.
202
203        Returns:
204            list[str]: A list of up to 'n' distinct non-null values from the specified column.
205        """
206        distinct_values = df[column].dropna().unique()
207        if len(distinct_values) > n:
208            return random.sample(list(distinct_values), n)
209        return list(distinct_values)

392        Unifies column descriptions in the provided data lake list using an LLM.

523                            file_info["data_themes"] = json_metadata.get('dataThemes', json_metadata.get('General', ''))
524                            file_info["coverage_sector"] = json_metadata.get('coverageSector', '')
525                            file_info["metadata_description"] = True
526
527                            # Extract column descriptions from JSON metadata
528                            for col_name in df.columns:
529                                if col_name in json_metadata and json_metadata[col_name] and json_metadata[col_name] != " - ":
530                                    metadata_column_descriptions[col_name] = json_metadata[col_name]
531
532                        except Exception as e:
533                            print(add_timestamp(color_text(f"Error loading/parsing metadata for {file_name} from {found_metadata_path}: {e}", "red")))
534                    else:
535                        print(add_timestamp(color_text(f"No matching metadata JSON found for {file_name} in {metadata_directory}", "yellow")))
536
537                # --- Process Columns for Statistics and Samples ---
538                for col_name in df.columns:
539                    values = df[col_name].dropna().astype(str).tolist()
540                    num_values = len(values)
541
542                    # Basic statistics
543                    if num_values > 0:
544                        lengths = [len(v) for v in values]
545                        max_length = max(lengths)
546                        min_length = min(lengths)
547                        avg_length = sum(lengths) / num_values
548                        counter = Counter(values)
549                        most_common_values = [val for val, _ in counter.most_common(min(20, num_values))] # Limit to 20
550                    else:
551                        max_length, min_length, avg_length = 0, 0, 0
552                        most_common_values = []
553
554                    column_dict = {
555                        "column_name": col_name,
556                        "header": col_name, # Original header
557                        "values_sample": self._get_distinct_sample_values(df, col_name, sample_size),
558                        "value_stats": {
559                            "num_values": num_values,
560                            "max_length": max_length,
561                            "min_length": min_length,
562                            "avg_length": round(avg_length, 2),
563                            "most_common_values": most_common_values
564                        },
565                        "metadata_description": metadata_column_descriptions.get(col_name) or None,
566                        "llm_description": None, # To be filled by LLM
567                        "semantic_annotation": None, # To be filled by SemanticAnnotator later (label)
568                    }
569                    file_info["columns"].append(column_dict)

575                    llm_response = self._get_llm_description(llm_prompt_truncated)

578                    file_info["llm_spatial_coverage"] = llm_response.get("spatial_coverage", "")
579
580                    for col_dict in file_info["columns"]:
581                        col_name = col_dict["column_name"]
582                        col_dict["llm_description"] = llm_response["column_descriptions"].get(col_name, "")
583                
584                data_lake_list.append(file_info)
585                
586            except Exception as e:
587                print(add_timestamp(color_text(f"Error processing file {file_name}: {e}", "red")))
