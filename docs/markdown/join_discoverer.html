 37    def _truncate_text(self,
 38        text: str,
 39        max_tokens: int,
 40        model_name: str = 'text-embedding-ada-002'
 41    ) -> str:
 42        """
 43        Private helper method to truncate text to fit within the token limit for the embedding model.
 44
 45        Args:
 46            text (str): The text to truncate.
 47            max_tokens (int): The maximum number of tokens allowed for the embedding model.
 48            model_name (str): The name of the embedding model to use.
 49
 50        Returns:
 51            str: The truncated text, or the original text if it is already within the token limit.
 52        """
 53        if self.tokenizer:
 54            encoded_text = self.tokenizer.encode(text)
 55        else:
 56            try:
 57                encoding = tiktoken.encoding_for_model(model_name)
 58            except KeyError:
 59                encoding = tiktoken.get_encoding("cl100k_base")
 60            encoded_text = encoding.encode(text)
 61
 62        if len(encoded_text) > max_tokens:
 63            truncated_text = self.tokenizer.decode(encoded_text[:max_tokens])
 64            return truncated_text
 65        return text

 67    def _prepare_embedding_prompt(self, column_data: dict, mode: str, max_tokens: int) -> Union[str, None]:
 68        """
 69        The prompt is built conditionally based on the available data and the specified mode.
 70        This method now combines information from semantic annotations, LLM descriptions,
 71        and column value statistics for a richer context.
 72
 73        Args:
 74            column_data (dict): A dictionary containing all information for a single column.
 75            mode (str): The mode to use for preparing the embedding prompt.
 76                        Options: 'semantic_mode', 'header_mode'.
 77            max_tokens (int): The maximum number of tokens allowed for the prompt.
 78
 79        Returns:
 80            Union[str, None]: The prepared prompt string, or None if no valid data is found.
 81        """
 82        # Extract data from the column dictionary
 83        header = column_data.get('header', column_data.get('column_name', ''))
 84        semantic_annotation = column_data.get('semantic_annotation', 'NA')
 85        llm_description = column_data.get('llm_description', '')
 86        value_stats = column_data.get('value_stats', {})
 87        attribute_examples = ", ".join(column_data.get('values_sample', []))
 88
 89        # Extract specific statistics
 90        num_values = value_stats.get('num_values', 0)
 91        max_length = value_stats.get('max_length', 0)
 92        min_length = value_stats.get('min_length', 0)
 93        avg_length = value_stats.get('avg_length', 0.0)
 94        most_common_values = value_stats.get('most_common_values', [])
 95
 96        text_parts = []
 97        if mode == 'semantic_mode' and semantic_annotation != 'NA':
 98            try:
 99                # Assuming format is "class_name.attribute"
100                class_name, _ = semantic_annotation.split('.', 1)
101            except ValueError:
102                class_name = 'NA'
103
104            prompt_line = f"The attribute with header: '{header}' and semantic annotation: '{semantic_annotation}', belongs to instances of type '{class_name}'"
105            if llm_description:
106                prompt_line += f" and is described as: '{llm_description}'."
107            else:
108                prompt_line += "."
109            text_parts.append(prompt_line)
110
111        elif mode == 'header_mode':
112            prompt_line = f"The attribute with header: '{header}'"
113            if llm_description:
114                prompt_line += f" is described as: '{llm_description}'."
115            else:
116                prompt_line += "."
117            text_parts.append(prompt_line)
118
119        # Add value examples if available
120        if attribute_examples:
121            text_parts.append(f"Examples of values for this attribute include: '{attribute_examples}'.")
122
123        # Add column statistics if available
124        if num_values > 0:
125            text_parts.append(f"The dataset column for '{header}' contains {num_values} entries.")
126
127        if max_length > 0 and min_length > 0:
128            text_parts.append("Key statistics for the column:")
129            text_parts.append(f"- Maximum value length: {max_length} characters.")
130            text_parts.append(f"- Minimum value length: {min_length} characters.")
131            text_parts.append(f"- Average value length: {avg_length:.1f} characters.")
132
133        # Add most frequent values if available
134        if most_common_values:
135            text_parts.append("Top 20 most frequent values in the column:")
136            text_parts.append(", ".join([str(v) for v in most_common_values[:20]]))
137
138        # Join and truncate the final prompt
139        final_prompt = "\n".join([part for part in text_parts if part])
140
141        if final_prompt:
142            return self._truncate_text(final_prompt, max_tokens)
143        else:
144            return None

167    def generate_embeddings(self,
168        data_lake_list: Union[list[dict], str],
169        prompt_mode: str = 'semantic_mode',
170        embedding_model: str = 'text-embedding-3-small',
171        output_directory: str | None = None
172    ) -> list[dict]:
173        """
174        Generates a list of embeddings for each column in the data lake representation.
175
176        Args:
177            data_lake_list (list[dict] | str): The list of dictionaries representing the data lake,
178                                            as generated by DataLoader.load_and_describe_datalake.
179            prompt_mode (str): One of 'semantic_mode' or 'data_profiling_mode'. See _prepare_embedding_prompt
180                                            for more information.
181            embedding_model (str): The name of the OpenAI embedding model to use.
182            output_directory (str | None): If specified, the generated embeddings will be saved to a file
183                                        in this directory, named "embeddings.json".
184
185        Returns:
186            list[dict]: A list of dictionaries, each containing the embedding for a column and its file name,
187                        as well as the semantic annotation for the column (if available).
188        """
189        if not data_lake_list:
190            print(add_timestamp(color_text("No data lake information provided for schema pruning.", "yellow")))
191            return None
192            
193        if isinstance(data_lake_list, str):
194            if not os.path.exists(data_lake_list):
195                print(add_timestamp(color_text(f"Error: Data lake JSON file not found at {data_lake_list}", "red")))
196            data_lake_list = load_datalake_json(data_lake_list)
197        
198        embeddings_list = []
199        max_prompt_tokens = 8191 if 'text-embedding-3-large' in embedding_model else 2048
200
201        total_columns = sum(len(file_info['columns']) for file_info in data_lake_list)
202
203        with tqdm(total=total_columns, desc=add_timestamp("Generating Embeddings")) as pbar:
204            for file_info in data_lake_list:
205                for column_dict in file_info['columns']:
206                    full_column_name = f"{file_info['file_name']}:{column_dict['column_name']}"
207
208                    prompt_text = self._prepare_embedding_prompt(
209                        column_data=column_dict,
210                        mode=prompt_mode,
211                        max_tokens=max_prompt_tokens
212                    )
213
214                    if prompt_text is None:
215                        pbar.update(1)
216                        continue
217
218                    embedding = self._get_embedding(prompt_text, model=embedding_model)
219
220                    if embedding is not None:
221                        embeddings_list.append({
222                            'column_name': full_column_name,
223                            'embedding': embedding,
224                            'semantic_annotation': column_dict.get('semantic_annotation', 'NA')
225                        })
226
227                    pbar.update(1)
228                    time.sleep(0.1)
229
230        print(add_timestamp(color_text(f"Finished generating embeddings for {len(embeddings_list)} columns.", "green")))
231        
232        if output_directory:
233            try:
234                create_directory_if_not_exists(output_directory)
235                output_file = os.path.join(output_directory, "embeddings.json")
236                with open(output_file, 'w', encoding='utf-8') as f:
237                    json.dump(embeddings_list, f, indent=2)
238                print(add_timestamp(color_text(f"Embeddings saved to: {output_file}", "green")))
239            except Exception as e:
240                print(add_timestamp(color_text(f"Error saving embeddings to {output_file}: {e}", "red")))
241        
242        return embeddings_list

244    def _compute_distances(self,
245        embeddings: list[dict],
246        output_directory
247    ) -> pd.DataFrame:
248        """
249        Compute cosine similarity, euclidean distance, and ANNS distances between all pairs of column embeddings.
250
251        Args:
252            embeddings (list[dict]): A list of dictionaries, each containing the embedding for a column and its file name,
253                                     as well as the semantic annotation for the column (if available).
254            output_directory (str): The directory where the output files will be saved.
255
256        Returns:
257            pd.DataFrame: A DataFrame containing the distances and similarities between all pairs of columns.
258        """
259        column_names = [d['column_name'] for d in embeddings]
260        embedding_matrix = np.array([d['embedding'] for d in embeddings])
261
262        # --- 1. Calculate Cosine and Euclidean Distances ---
263        print(add_timestamp(color_text("Calculating cosine similarity and euclidean distances...", 'cyan')))
264        cosine_sim_matrix = cosine_similarity(embedding_matrix)
265        euclidean_dist_matrix = euclidean_distances(embedding_matrix)
266
267        # --- 2. Calculate ANNS Distances (using FAISS) ---
268        print(add_timestamp(color_text("Calculating ANNS distances...", 'cyan')))
269        try:
270            d = embedding_matrix.shape[1]
271            index = faiss.IndexFlatL2(d)
272            index.add(embedding_matrix.astype('float32'))
273            distances_anns, indices_anns = index.search(embedding_matrix.astype('float32'), len(column_names))
274
275        except ImportError:
276            print(add_timestamp(color_text("FAISS library not found. ANNS distances will be skipped.", "red")))
277            distances_anns, indices_anns = None, None
278        except Exception as e:
279            print(add_timestamp(color_text(f"Error calculating ANNS distances: {e}. ANNS distances will be skipped.", "red")))
280            distances_anns, indices_anns = None, None
281        
282        # Export distances
283        distances = []
284        for i in range(len(column_names)):
285            for j in range(len(column_names)):
286                if i != j: # Exclude self-comparison
287                    distances.append({
288                        "column_1": column_names[i],
289                        "column_2": column_names[j],
290                        "cosine_similarity": cosine_sim_matrix[i, j],
291                        "euclidean_distance": euclidean_dist_matrix[i, j],
292                        "distance_anns": distances_anns[i, j] if distances_anns is not None else None
293                    })
294        sorted_distances = sorted(distances, key=lambda x: (x['euclidean_distance'], -x['cosine_similarity'], x['distance_anns']))
295        
296        if output_directory:
297            try:
298                create_directory_if_not_exists(output_directory)
299                output_file = os.path.join(output_directory, "distances.csv")
300                df_distances = pd.DataFrame(sorted_distances)
301                df_distances.to_csv(output_file, index=False, encoding='utf-8')
302                print(add_timestamp(color_text(f"Distances saved to: {output_file}", "green")))
303            except Exception as e:
304                print(add_timestamp(color_text(f"Error saving distances to {output_file}: {e}", "red")))
305        
306        return df_distances

308    def _prepare_nodes(self,
309        embeddings: list[dict],
310        output_directory: str | None = None
311    ) -> pd.DataFrame:
312        """
313        Prepare nodes DataFrame for export to Neo4j. This function takes the input embeddings,
314        drops the "semantic_annotation" column, splits the "column_name" column into "docID" and "colname",
315        removes duplicates by "docID" and "colname", pivots the DataFrame to have one row per file
316        and one column per CSV column name, cleans the column names, and renames the "docID" column
317        to the Neo4j-compatible node ID. The resulting DataFrame is then saved to a CSV file in
318        the specified output directory.
319
320        Args:
321            embeddings (list[dict]): The list of dictionaries with 'column_name', 'embedding', and 'semantic_annotation'.
322            output_directory (str): The directory where output files will be saved.
323
324        Returns:
325            pd.DataFrame: The prepared nodes DataFrame.
326        """
327        print(add_timestamp(color_text("Generating Neo4j nodes CSV...", 'cyan')))
328        nodes = embeddings.copy()
329        nodes = pd.DataFrame(nodes)
330        nodes = nodes.drop(columns=["semantic_annotation"])
331        nodes[["docID", "colname"]] = nodes["column_name"].str.split(":", expand=True) # Split 'column_name' into 'docID' (filename) and 'colname' (column header)
332        nodes = nodes.drop(columns=["column_name"])
333        nodes = nodes.drop_duplicates(subset=["docID", "colname"]) # Remove duplicates by 'docID' and 'colname'
334        pivoted = nodes.pivot(index="docID", columns="colname", values="embedding").reset_index() # Pivot: one row per file, one column per CSV column name
335        pivoted.columns = pivoted.columns.str.replace(":", "_", regex=False) # Clean column names: replace ':' with '_' and empty names with '_'
336        pivoted.columns = pivoted.columns.str.replace("^$", "_", regex=True)
337        pivoted = pivoted.rename(columns={"docID": "docID:ID"}) # Rename 'docID' to Neo4j-compatible node ID
338        
339        if output_directory:
340            try:
341                create_directory_if_not_exists(output_directory)
342                output_file = os.path.join(output_directory, "nodes.csv")
343                pivoted.to_csv(output_file, index=False, encoding='utf-8')
344                print(add_timestamp(color_text(f"Neo4j nodes saved to: {output_file}", "green")))
345            except Exception as e:
346                print(add_timestamp(color_text(f"Error saving neo4j nodes to {output_file}: {e}", "red")))
347        
348        return pivoted

350    def _prepare_edges(self,
351        df: pd.DataFrame,
352        cosine_sim_threshold: float = 0.5,
353        anns_threshold: float = 0.2,
354        output_directory: str | None = None
355    ) -> pd.DataFrame:    
356        """
357        Prepare edges DataFrame for export to Neo4j. This function takes the input DataFrame, 
358        splits the "column_1" and "column_2" columns into "docID" and "colname", inverts the 
359        direction of the edges and removes duplicates, cleans the column names, and renames the 
360        "docID" column to the Neo4j-compatible node ID. The resulting DataFrame is then saved to 
361        a CSV file in the specified output directory.
362
363        Args:
364            df (pd.DataFrame): The input DataFrame with 'column_1', 'column_2', 'cosine_similarity', and 'distance_anns'.
365            output_directory (str): The directory where output files will be saved.
366
367        Returns:
368            pd.DataFrame: The prepared edges DataFrame.
369        """
370        df[['f1', 'c1']] = df['column_1'].str.split(':', expand=True)
371        df[['f2', 'c2']] = df['column_2'].str.split(':', expand=True)
372        df['c1'] = df['c1'].str.replace(":", "_", regex=False).str.replace("^$", "_", regex=True)
373        df['c2'] = df['c2'].str.replace(":", "_", regex=False).str.replace("^$", "_", regex=True)
374        df[':TYPE'] = df['c1'] + " -> " + df['c2']
375        df = df.rename(columns={'f1': ':START_ID', 'f2': ':END_ID'})
376        df = df.drop(columns=['column_1', 'column_2', 'c1', 'c2'])
377        
378        def invert(df):
379            """
380            Inverts the direction of the edges in the given DataFrame.
381            """
382            inv = df.copy()
383            inv = inv.rename(columns={':START_ID': ':END_ID', ':END_ID': ':START_ID'})
384            inv[':TYPE'] = inv[':TYPE'].apply(lambda x: ' -> '.join(x.split(' -> ')[::-1]))
385            return inv
386        
387        df_all = pd.concat([df, invert(df)], ignore_index=True).drop_duplicates()
388        
389        edges = df_all[
390            (df_all[':START_ID'] != df_all[':END_ID']) &
391            (df_all['cosine_similarity'] >= cosine_sim_threshold) &
392            (df_all['distance_anns'] <= anns_threshold)
393        ].copy()
394
395        edges['pair_type'] = edges.apply(
396            lambda r: (tuple(sorted([r[':START_ID'], r[':END_ID']])),
397                ' -> '.join(sorted(r[':TYPE'].split(' -> ')))
398                if r[':START_ID'] > r[':END_ID'] else r[':TYPE']),
399            axis=1)
400        edges =  edges.drop_duplicates('pair_type').drop(columns=['pair_type'])
401        
402        if output_directory:
403            try:
404                create_directory_if_not_exists(output_directory)
405                output_file = os.path.join(output_directory, "edges.csv")
406                edges.to_csv(output_file, index=False, encoding='utf-8')
407                print(add_timestamp(color_text(f"Neo4j edges saved to: {output_file}", "green")))
408            except Exception as e:
409                print(add_timestamp(color_text(f"Error saving neo4j edges to {output_file}: {e}", "red")))
410        
411        return edges

413    def compute_distances_and_export_neo4j(self,
414        embeddings: Union[list[dict], str],
415        cosine_sim_threshold: float = 0.5,
416        anns_threshold: float = 0.2,
417        output_directory: str | None = None,
418    ) -> tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame]:
419        """
420        Computes cosine similarity, euclidean distance, and ANNS distances between all pairs of column embeddings, 
421        and exports the results as CSV files for Neo4j nodes and edges.
422
423        Args:
424            embeddings: A list of dictionaries, each containing the embedding for a column and its file name,
425                        as well as the semantic annotation for the column (if available). Alternatively, a string
426                        path to a JSON file containing the embeddings.
427            cosine_sim_threshold: The minimum cosine similarity required for a pair of columns to be considered joinable.
428            anns_threshold: The maximum ANNS distance required for a pair of columns to be considered joinable.
429            output_directory: The directory where the output CSV files will be saved.
430
431        Returns:
432            tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame]: The computed distances DataFrame, the nodes DataFrame,
433                and the edges DataFrame.
434        """
435        if not embeddings:
436            print(add_timestamp(color_text("No embeddings provided. Skipping distance calculation and export.", "yellow")))
437            return None
438            
439        if isinstance(embeddings, str):
440            if not os.path.exists(embeddings):
441                print(add_timestamp(color_text(f"Error: embeddings JSON file not found at {embeddings}", "red")))
442            with open(embeddings, "r") as f:
443                embeddings = json.load(f)
444
445        # --- Compute Distances ---
446        df_distances = self._compute_distances(embeddings, output_directory)
447
448        # --- Prepare Neo4j Nodes CSV ---
449        df_nodes = self._prepare_nodes(embeddings, output_directory)
450        
451        # --- Prepare Neo4j Edges CSV ---
452        df_edges = self._prepare_edges(df_distances, cosine_sim_threshold, anns_threshold, output_directory)
453        
454        return df_distances, df_nodes, df_edges

245    def _compute_distances(self,
246        embeddings: list[dict],
247        output_directory
248    ) -> pd.DataFrame:
249        """
250        Compute cosine similarity, euclidean distance, and ANNS distances between all pairs of column embeddings.
251
252        Args:
253            embeddings (list[dict]): A list of dictionaries, each containing the embedding for a column and its file name,
254                                     as well as the semantic annotation for the column (if available).
255            output_directory (str): The directory where the output files will be saved.
256
257        Returns:
258            pd.DataFrame: A DataFrame containing the distances and similarities between all pairs of columns.
259        """
260        column_names = [d['column_name'] for d in embeddings]
261        embedding_matrix = np.array([d['embedding'] for d in embeddings])
262
263        # --- 1. Calculate Cosine and Euclidean Distances ---
264        print(add_timestamp(color_text("Calculating cosine similarity and euclidean distances...", 'cyan')))
265        cosine_sim_matrix = cosine_similarity(embedding_matrix)
266        euclidean_dist_matrix = euclidean_distances(embedding_matrix)
267
268        # --- 2. Calculate ANNS Distances (using FAISS) ---
269        print(add_timestamp(color_text("Calculating ANNS distances...", 'cyan')))
270        try:
271            d = embedding_matrix.shape[1]
272            index = faiss.IndexFlatL2(d)
273            index.add(embedding_matrix.astype('float32'))
274            distances_anns, indices_anns = index.search(embedding_matrix.astype('float32'), len(column_names))
275
276        except ImportError:
277            print(add_timestamp(color_text("FAISS library not found. ANNS distances will be skipped.", "red")))
278            distances_anns, indices_anns = None, None
279        except Exception as e:
280            print(add_timestamp(color_text(f"Error calculating ANNS distances: {e}. ANNS distances will be skipped.", "red")))
281            distances_anns, indices_anns = None, None
282        
283        # Export distances
284        distances = []
285        for i in range(len(column_names)):
286            for j in range(len(column_names)):
287                if i != j: # Exclude self-comparison
288                    distances.append({
289                        "column_1": column_names[i],
290                        "column_2": column_names[j],
291                        "cosine_similarity": cosine_sim_matrix[i, j],
292                        "euclidean_distance": euclidean_dist_matrix[i, j],
293                        "distance_anns": distances_anns[i, j] if distances_anns is not None else None
294                    })
295        sorted_distances = sorted(distances, key=lambda x: (x['euclidean_distance'], -x['cosine_similarity'], x['distance_anns']))
296        
297        if output_directory:
298            try:
299                create_directory_if_not_exists(output_directory)
300                output_file = os.path.join(output_directory, "distances.csv")
301                df_distances = pd.DataFrame(sorted_distances)
302                df_distances.to_csv(output_file, index=False, encoding='utf-8')
303                print(add_timestamp(color_text(f"Distances saved to: {output_file}", "green")))
304            except Exception as e:
305                print(add_timestamp(color_text(f"Error saving distances to {output_file}: {e}", "red")))
306        
307        return df_distances

309    def _prepare_nodes(self,
310        embeddings: list[dict],
311        output_directory: str | None = None
312    ) -> pd.DataFrame:
313        """
314        Prepare nodes DataFrame for export to Neo4j. This function takes the input embeddings,
315        drops the "semantic_annotation" column, splits the "column_name" column into "docID" and "colname",
316        removes duplicates by "docID" and "colname", pivots the DataFrame to have one row per file
317        and one column per CSV column name, cleans the column names, and renames the "docID" column
318        to the Neo4j-compatible node ID. The resulting DataFrame is then saved to a CSV file in
319        the specified output directory.
320
321        Args:
322            embeddings (list[dict]): The list of dictionaries with 'column_name', 'embedding', and 'semantic_annotation'.
323            output_directory (str): The directory where output files will be saved.
324
325        Returns:
326            pd.DataFrame: The prepared nodes DataFrame.
327        """
328        print(add_timestamp(color_text("Generating Neo4j nodes CSV...", 'cyan')))
329        nodes = embeddings.copy()
330        nodes = pd.DataFrame(nodes)
331        nodes = nodes.drop(columns=["semantic_annotation"])
332        nodes[["docID", "colname"]] = nodes["column_name"].str.split(":", expand=True) # Split 'column_name' into 'docID' (filename) and 'colname' (column header)
333        nodes = nodes.drop(columns=["column_name"])
334        nodes = nodes.drop_duplicates(subset=["docID", "colname"]) # Remove duplicates by 'docID' and 'colname'
335        pivoted = nodes.pivot(index="docID", columns="colname", values="embedding").reset_index() # Pivot: one row per file, one column per CSV column name
336        pivoted.columns = pivoted.columns.str.replace(":", "_", regex=False) # Clean column names: replace ':' with '_' and empty names with '_'
337        pivoted.columns = pivoted.columns.str.replace("^$", "_", regex=True)
338        pivoted = pivoted.rename(columns={"docID": "docID:ID"}) # Rename 'docID' to Neo4j-compatible node ID
339        
340        if output_directory:
341            try:
342                create_directory_if_not_exists(output_directory)
343                output_file = os.path.join(output_directory, "nodes.csv")
344                pivoted.to_csv(output_file, index=False, encoding='utf-8')
345                print(add_timestamp(color_text(f"Neo4j nodes saved to: {output_file}", "green")))
346            except Exception as e:
347                print(add_timestamp(color_text(f"Error saving neo4j nodes to {output_file}: {e}", "red")))
348        
349        return pivoted

351    def _prepare_edges(self,
352        df: pd.DataFrame,
353        cosine_sim_threshold: float = 0.5,
354        anns_threshold: float = 0.2,
355        output_directory: str | None = None
356    ) -> pd.DataFrame:    
357        """
358        Prepare edges DataFrame for export to Neo4j. This function takes the input DataFrame, 
359        splits the "column_1" and "column_2" columns into "docID" and "colname", inverts the 
360        direction of the edges and removes duplicates, cleans the column names, and renames the 
361        "docID" column to the Neo4j-compatible node ID. The resulting DataFrame is then saved to 
362        a CSV file in the specified output directory.
363
364        Args:
365            df (pd.DataFrame): The input DataFrame with 'column_1', 'column_2', 'cosine_similarity', and 'distance_anns'.
366            output_directory (str): The directory where output files will be saved.
367
368        Returns:
369            pd.DataFrame: The prepared edges DataFrame.
370        """
371        df[['f1', 'c1']] = df['column_1'].str.split(':', expand=True)
372        df[['f2', 'c2']] = df['column_2'].str.split(':', expand=True)
373        df['c1'] = df['c1'].str.replace(":", "_", regex=False).str.replace("^$", "_", regex=True)
374        df['c2'] = df['c2'].str.replace(":", "_", regex=False).str.replace("^$", "_", regex=True)
375        df[':TYPE'] = df['c1'] + " -> " + df['c2']
376        df = df.rename(columns={'f1': ':START_ID', 'f2': ':END_ID'})
377        df = df.drop(columns=['column_1', 'column_2', 'c1', 'c2'])
378        
379        def invert(df):
380            """
381            Inverts the direction of the edges in the given DataFrame.
382            """
383            inv = df.copy()
384            inv = inv.rename(columns={':START_ID': ':END_ID', ':END_ID': ':START_ID'})
385            inv[':TYPE'] = inv[':TYPE'].apply(lambda x: ' -> '.join(x.split(' -> ')[::-1]))
386            return inv
387        
388        df_all = pd.concat([df, invert(df)], ignore_index=True).drop_duplicates()
389        
390        edges = df_all[
391            (df_all[':START_ID'] != df_all[':END_ID']) &
392            (df_all['cosine_similarity'] >= cosine_sim_threshold) &
393            (df_all['distance_anns'] <= anns_threshold)
394        ].copy()
395
396        edges['pair_type'] = edges.apply(
397            lambda r: (tuple(sorted([r[':START_ID'], r[':END_ID']])),
398                ' -> '.join(sorted(r[':TYPE'].split(' -> ')))
399                if r[':START_ID'] > r[':END_ID'] else r[':TYPE']),
400            axis=1)
401        edges =  edges.drop_duplicates('pair_type').drop(columns=['pair_type'])
402        
403        if output_directory:
404            try:
405                create_directory_if_not_exists(output_directory)
406                output_file = os.path.join(output_directory, "edges.csv")
407                edges.to_csv(output_file, index=False, encoding='utf-8')
408                print(add_timestamp(color_text(f"Neo4j edges saved to: {output_file}", "green")))
409            except Exception as e:
410                print(add_timestamp(color_text(f"Error saving neo4j edges to {output_file}: {e}", "red")))
411        
412        return edges
