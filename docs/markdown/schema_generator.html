 30    def _extract_classes_from_columns(self,
 31        columns
 32    ) -> dict:
 33        """
 34        Extracts classes and their attributes from a list of column names.
 35
 36        This method processes a list of column names, identifying classes based on
 37        the format `class.attribute`. If a column name follows this format, the
 38        attribute is added to the corresponding class. If a column name does not
 39        have a class prefix, it is added to a "Document" class.
 40
 41        Args:
 42            columns (list of str): The list of column names to process.
 43
 44        Returns:
 45            dict: A dictionary where keys are class names and values are lists of attributes.
 46        """
 47        classes = defaultdict(list)
 48        
 49        for column in columns:
 50            parts = column.split('.')
 51            if len(parts) > 1:
 52                # If format is class.attribute, add to that class
 53                class_name = parts[0]
 54                attribute = '.'.join(parts[1:])
 55                classes[class_name].append(attribute)
 56            else:
 57                # If no class prefix, add to a general Document class
 58                classes["Document"].append(column)
 59        
 60        return classes

 62    def generate_linkml_schema(self,
 63        data_lake_list: Union[list[dict], str],
 64        output_directory: str | None = None
 65    ) -> dict:
 66        """
 67        Generates a LinkML schema from a processed data lake representation.
 68        
 69        This method collects column data, unifies descriptions for common columns,
 70        and then generates the LinkML YAML schema using an LLM.
 71        
 72        Args:
 73            data_lake_list (list[dict] | str): The list of dictionaries representing the data lake,
 74                                            as generated by DataLoader.load_and_describe_datalake.
 75            output_directory (str | None): If specified, the generated LinkML schema will be saved to a file
 76                                        in this directory, named "linkml_schema.yaml".
 77        
 78        Returns:
 79            dict | None: The generated LinkML schema in dictionary format, or None if an error occurs.
 80        """
 81        if not data_lake_list:
 82            print(add_timestamp(color_text("No data lake information provided to generate schema.", "yellow")))
 83            return None
 84        
 85        if isinstance(data_lake_list, str):
 86            if not os.path.exists(data_lake_list):
 87                print(add_timestamp(color_text(f"Error: Data lake JSON file not found at {data_lake_list}", "red")))
 88            data_lake_list = load_datalake_json(data_lake_list)
 89
 90        column_names = []
 91        all_columns = defaultdict(lambda: {"description": [], "examples": set()})
 92        for file_info in data_lake_list:
 93            for col_dict in file_info['columns']:
 94                col_name = col_dict['column_name']
 95                column_names.append(col_name)
 96                all_columns[col_name]["description"] = col_dict.get('llm_description', '')
 97                all_columns[col_name]["examples"] = col_dict.get('values_sample', [])
 98        column_names = set(column_names)
 99        classes = self._extract_classes_from_columns(column_names)
100
101        # --- Generate LinkML Schema using LLM ---
102        prompt = f"""Create a LinkML schema for a Eurostat dataset with a single "Document" class that inherits from "NamedEntity".
103
104IDENTIFIED CLASSES AND THEIR ATTRIBUTES:
105"""
106        for class_name, attributes in classes.items():
107            prompt += f"\nCLASS: {class_name}\nATTRIBUTES: {', '.join(attributes)}\n"
108        
109        prompt += "\nDETAILED COLUMN DATA (description and examples):\n"
110        for column, data in all_columns.items():
111            description = data["description"]
112            examples = data["examples"]
113            prompt += f"\n{column}:\n  Description: {description}\n  Examples: {', '.join(examples)}"
114        
115        prompt += """
116
117TASK:
118Create a comprehensive LinkML schema that includes all identified classes and their attributes.
119
120RULES:
1211. Follow the LinkML schema format as shown in the example below
1222. Create a class for each identified class in the data
1233. All classes should inherit from "NamedEntity"
1244. For attributes that don't have a class prefix (like 'attribute' vs 'class.attribute'), 
125   add them to a "Document" class that also inherits from "NamedEntity"
1265. Use appropriate data types for each attribute based on the examples provided (string, integer, float, etc.)
1276. Include clear descriptions for each attribute
1287. Include examples for each attribute using the provided examples
1298. Group related attributes together in the YAML structure
1309. Use standard LinkML prefixes and imports
131
132EXAMPLE FORMAT:
133```yaml
134id: https://w3id.org/schema
135name: dataset-schema
136title: Dataset Schema
137description: >-
138  A schema for a dataset.
139prefixes:
140  linkml: https://w3id.org/linkml/
141  core: http://w3id.org/ontogpt/core/
142
143default_range: string
144
145imports:
146  - linkml:types
147  - core
148
149classes:
150  Document:
151    is_a: NamedEntity
152    description: A document representing a dataset
153    attributes:
154      attribute1:
155        description: Description of attribute1
156        range: string
157        examples: example1, example2
158  
159  Gene:
160    is_a: NamedEntity
161    description: A gene entity
162    attributes:
163      type:
164        description: The type of gene
165        range: string
166        examples: protein-coding, ncRNA
167      symbol:
168        description: The gene symbol
169        range: string
170        examples: BRCA1, TP53
171```
172
173RESPONSE FORMAT:
174{
175    "schema": "Your complete LinkML schema in YAML format here (without the ```yaml and ``` markers)"
176}
177"""
178        print(add_timestamp(color_text("Generating LinkML schema with LLM...", "yellow")))
179        try:
180            response = self.openai_client.chat.completions.create(
181                model="gpt-4o", # Can be made configurable
182                messages=[
183                    {"role": "system", "content": "You are a data modeling expert specializing in LinkML schemas. You create well-structured, comprehensive schemas following LinkML conventions. Always respond in JSON format."},
184                    {"role": "user", "content": prompt}
185                ],
186                temperature=0.1,
187                response_format={"type": "json_object"}
188            )
189
190            result = json.loads(response.choices[0].message.content)
191            schema_yaml = result.get("schema", "")
192
193            # Basic validation and correction for 'is_a: NamedEntity'
194            try:
195                yaml_data = yaml.safe_load(schema_yaml)
196                if "classes" not in yaml_data or "Document" not in yaml_data["classes"]:
197                    print(add_timestamp(color_text("Warning: Schema does not contain a 'Document' class. Attempting to add it.", "yellow")))
198                    if "classes" not in yaml_data:
199                        yaml_data["classes"] = {}
200                    yaml_data["classes"]["Document"] = {"is_a": "NamedEntity", "description": "A document representing a dataset"}
201                    # If attributes were generated but not under Document, try to move them
202                    if "attributes" in yaml_data and "Document" not in yaml_data["classes"]:
203                        yaml_data["classes"]["Document"]["attributes"] = yaml_data.pop("attributes")
204                elif "is_a" not in yaml_data["classes"]["Document"] or yaml_data["classes"]["Document"]["is_a"] != "NamedEntity":
205                    print(add_timestamp(color_text("Warning: Correcting 'Document' class 'is_a' to 'NamedEntity'.", "yellow")))
206                    yaml_data["classes"]["Document"]["is_a"] = "NamedEntity"
207
208                schema_yaml = yaml.dump(yaml_data, default_flow_style=False, sort_keys=False, indent=2)
209            except Exception as e:
210                print(add_timestamp(color_text(f"Warning: Could not validate or correct schema structure: {e}", "red")))
211
212            print(add_timestamp(color_text("Successfully generated LinkML schema.", "green")))
213            
214            if output_directory:
215                try:
216                    create_directory_if_not_exists(output_directory)
217                    output_file = os.path.join(output_directory, "linkml_schema.yaml")
218                    with open(output_file, 'w', encoding='utf-8') as f:
219                        f.write(schema_yaml)
220                    print(add_timestamp(color_text(f"LinkML schema saved to: {output_file}", "green")))
221                except Exception as e:
222                    print(add_timestamp(color_text(f"Error saving LinkML schema: {e}", "red")))
223            
224            return yaml_data
225
226        except Exception as e:
227            print(add_timestamp(color_text(f"Error generating LinkML schema: {e}", "red")))
228            return None

230    def _classify_document_classes(self,
231        file_info: dict,
232        schema_descriptions: dict
233    ) -> dict | None:
234        """
235        Creates a prompt and gets classification from LLM to identify the most relevant
236        schema classes for the entire document.
237
238        Args:
239            file_info (dict): A dictionary representing a single file from the data lake list.
240            schema_descriptions (dict): Schema class descriptions (only top-level classes).
241
242        Returns:
243            dict | None: Classification results, or None if an error occurs.
244        """
245        prompt = f"""Analyze this dataset as a whole document and identify which schema classes (maximum 3) best characterize its content and purpose.
246
247FILE INFORMATION:
248- File name: {file_info['file_name']}"""
249
250        if file_info['original_headers']:
251            prompt += f"\n- Headers: {', '.join(file_info['original_headers'])}"
252
253        prompt += "\n\nCOLUMN SAMPLES (first 5 samples per column):"
254
255        for col_dict in file_info['columns']:
256            # Use values_sample from the column dict
257            prompt += f"\n{col_dict['column_name']}: {col_dict['values_sample'][:5]}"
258
259        prompt += "\n\nSCHEMA CLASSES:"
260        for class_name, description in schema_descriptions.items():
261            prompt += f"\n{class_name}: {description}"
262
263        prompt += """
264
265TASK:
266Identify the most relevant schema classes (maximum 3) that best characterize this document as a whole.
267
268RESPONSE FORMAT:
269{
270    "file_name": "example.tsv",
271    "relevant_classes": ["Class1", "Class2"],  // Maximum 3 classes, ordered by relevance
272    "reasoning": [
273        "1. Why this document primarily deals with Class1...",
274        "2. Why Class2 is also relevant for this document...",
275        "3. Why other classes are less relevant or not applicable..."
276    ]
277}
278
279RULES:
280- Consider the document as a whole, not individual columns.
281- Select maximum 3 classes that best describe the document's domain and purpose.
282- Base your decision on:
283    * The overall content and purpose of the dataset.
284    * The types of entities being described or linked.
285    * The domain context suggested by the file name and data.
286- Provide clear explanations for why these classes are most relevant.
287- Explain why other classes were not selected.
288- Only select classes that are strongly represented in the document.
289"""
290        try:
291            response = self.openai_client.chat.completions.create(
292                model="gpt-4o", # Can be made configurable
293                messages=[
294                    {"role": "system", "content": "You are a data scientist specialized in data classification. You always respond in the exact JSON format requested, with no additional text before or after."},
295                    {"role": "user", "content": prompt}
296                ],
297                temperature=0.1,
298                response_format={"type": "json_object"}
299            )
300            return json.loads(response.choices[0].message.content)
301        except Exception as e:
302            print(add_timestamp(color_text(f"Error getting classification for {file_info.get('file_name', '?')}: {e}", "red")))
303            return None

305    def prune_schema(self,
306        data_lake_list: Union[list[dict], str],
307        yaml_schema: Union[str, dict],
308        output_directory: str | None = None,
309    ) -> str | None:
310        """
311        Prunes the given LinkML schema (yaml_schema) to include only the classes
312        that are relevant to the datasets in the data lake (data_lake_list).
313
314        The pruning process involves the following steps:
315        1. Classify each dataset in the data lake using the ChatGPT API and the
316        class descriptions from the LinkML schema.
317        2. Collect the relevant classes across all datasets.
318        3. Create a new LinkML schema with only the relevant classes and their
319        attributes.
320
321        Args:
322            data_lake_list (list[dict] or str): The list of dictionaries representing
323                the data lake, or a path to a JSON file containing the data lake
324                information.
325            yaml_schema (str or dict): The LinkML schema as a dictionary, or a path
326                to a YAML file containing the LinkML schema.
327            output_directory (str, optional): The directory where the pruned schema
328                should be saved. If not provided, the pruned schema is returned as a
329                dictionary.
330
331        Returns:
332            str or None: The path to the pruned schema YAML file if output_directory
333                is provided, otherwise the pruned schema as a dictionary.
334        """
335        if not data_lake_list:
336            print(add_timestamp(color_text("No data lake information provided for schema pruning.", "yellow")))
337            return None
338        
339        if isinstance(data_lake_list, str):
340            if not os.path.exists(data_lake_list):
341                print(add_timestamp(color_text(f"Error: Data lake JSON file not found at {data_lake_list}", "red")))
342            data_lake_list = load_datalake_json(data_lake_list)
343        
344        if not yaml_schema:
345            print(add_timestamp(color_text("No data lake information provided for schema pruning.", "yellow")))
346            return None
347        
348        if isinstance(yaml_schema, str):
349            if not os.path.exists(yaml_schema):
350                print(add_timestamp(color_text(f"Error: Full YAML schema not found at {yaml_schema}", "red")))
351                return None
352            
353            with open(yaml_schema, 'r', encoding='utf-8') as f:
354                yaml_schema = yaml.safe_load(f)
355
356        print(add_timestamp(color_text("Starting schema pruning...", "yellow")))
357
358        # Load full schema descriptions (only top-level classes for pruning classification)
359        _, full_yaml_descriptions = yaml_parser(yaml_schema)
360        filtered_class_descriptions = {k: v for k, v in full_yaml_descriptions.items() if '.' not in k}
361
362        # Collect relevant classes across all datasets
363        all_relevant_classes = set()
364        file_pruning_classifications = [] # To store detailed classifications per file
365
366        for file_info in tqdm(data_lake_list, desc="Classifying datasets for pruning"):
367            classification = self._classify_document_classes(file_info, filtered_class_descriptions)
368            if classification and classification.get('relevant_classes'):
369                all_relevant_classes.update(classification['relevant_classes'])
370                file_pruning_classifications.append({
371                    "file_name": file_info['file_name'],
372                    "relevant_classes": classification['relevant_classes'],
373                    "reasoning": classification.get('reasoning', [])
374                })
375
376        if not all_relevant_classes:
377            print(add_timestamp(color_text("No relevant classes identified for any dataset. Pruned schema will be empty.", "yellow")))
378            return None
379
380        print(add_timestamp(color_text(f"Total unique relevant classes identified: {len(all_relevant_classes)}", "blue")))
381        print(add_timestamp(color_text(f"Relevant classes: {list(all_relevant_classes)}", "blue")))
382
383        pruned_schema_data = yaml_schema.copy()
384        pruned_schema_data['classes'] = {}
385
386        # Populate pruned schema with only the relevant classes and their attributes
387        if 'classes' in yaml_schema:
388            for class_name, class_details in yaml_schema['classes'].items():
389                if class_name in all_relevant_classes:
390                    pruned_schema_data['classes'][class_name] = class_details.copy()
391                    # Ensure inherited classes are also included if they are not explicitly relevant
392                    # This is a simplification; a full LinkML pruning would trace dependencies
393                    is_a = class_details.get('is_a')
394                    if is_a and is_a not in pruned_schema_data['classes'] and is_a in yaml_schema['classes']:
395                        pruned_schema_data['classes'][is_a] = yaml_schema['classes'][is_a].copy()
396
397        if not pruned_schema_data['classes']:
398            print(add_timestamp(color_text("No classes left after pruning. Output YAML will be minimal.", "yellow")))
399
400        # Save the pruned schema to a new YAML file
401        if output_directory:
402            try:
403                create_directory_if_not_exists(output_directory)
404                output_file = os.path.join(output_directory, "linkml_schema_pruned.yaml")
405                with open(output_file, 'w', encoding='utf-8') as f:
406                    yaml.dump(pruned_schema_data, f, indent=2, sort_keys=False)
407                print(add_timestamp(color_text(f"Pruned LinkML schema saved to: {output_file}", "green")))
408
409                
410            except Exception as e:
411                print(add_timestamp(color_text(f"Error saving pruned schema to {output_file}: {e}", "red")))
412                return None
413        
414        return pruned_schema_data

307        data_lake_list: Union[list[dict], str],
308        yaml_schema: Union[str, dict],
309        output_directory: str | None = None,
310    ) -> str | None:
311        """
312        Prunes the given LinkML schema (yaml_schema) to include only the classes
313        that are relevant to the datasets in the data lake (data_lake_list).
314
315        The pruning process involves the following steps:
316        1. Classify each dataset in the data lake using the ChatGPT API and the
317        class descriptions from the LinkML schema.
318        2. Collect the relevant classes across all datasets.
319        3. Create a new LinkML schema with only the relevant classes and their
320        attributes.
321
322        Args:
323            data_lake_list (list[dict] or str): The list of dictionaries representing
324                the data lake, or a path to a JSON file containing the data lake
325                information.
326            yaml_schema (str or dict): The LinkML schema as a dictionary, or a path
327                to a YAML file containing the LinkML schema.
328            output_directory (str, optional): The directory where the pruned schema
329                should be saved. If not provided, the pruned schema is returned as a
330                dictionary.
331
332        Returns:
333            str or None: The path to the pruned schema YAML file if output_directory
334                is provided, otherwise the pruned schema as a dictionary.
335        """
336        if not data_lake_list:
337            print(add_timestamp(color_text("No data lake information provided for schema pruning.", "yellow")))
338            return None
339        
340        if isinstance(data_lake_list, str):
341            if not os.path.exists(data_lake_list):
342                print(add_timestamp(color_text(f"Error: Data lake JSON file not found at {data_lake_list}", "red")))
343            data_lake_list = load_datalake_json(data_lake_list)
344        
345        if not yaml_schema:
346            print(add_timestamp(color_text("No data lake information provided for schema pruning.", "yellow")))
347            return None
348        
349        if isinstance(yaml_schema, str):
350            if not os.path.exists(yaml_schema):
351                print(add_timestamp(color_text(f"Error: Full YAML schema not found at {yaml_schema}", "red")))
352                return None
353            
354            with open(yaml_schema, 'r', encoding='utf-8') as f:
355                yaml_schema = yaml.safe_load(f)
356
357        print(add_timestamp(color_text("Starting schema pruning...", "yellow")))
358
359        # Load full schema descriptions (only top-level classes for pruning classification)
360        _, full_yaml_descriptions = yaml_parser(yaml_schema)
361        filtered_class_descriptions = {k: v for k, v in full_yaml_descriptions.items() if '.' not in k}
362
363        # Collect relevant classes across all datasets
364        all_relevant_classes = set()
365        file_pruning_classifications = [] # To store detailed classifications per file
366
367        for file_info in tqdm(data_lake_list, desc="Classifying datasets for pruning"):
368            classification = self._classify_document_classes(file_info, filtered_class_descriptions)
369            if classification and classification.get('relevant_classes'):
370                all_relevant_classes.update(classification['relevant_classes'])
371                file_pruning_classifications.append({
372                    "file_name": file_info['file_name'],
373                    "relevant_classes": classification['relevant_classes'],
374                    "reasoning": classification.get('reasoning', [])
375                })
376
377        if not all_relevant_classes:
378            print(add_timestamp(color_text("No relevant classes identified for any dataset. Pruned schema will be empty.", "yellow")))
379            return None
380
381        print(add_timestamp(color_text(f"Total unique relevant classes identified: {len(all_relevant_classes)}", "blue")))
382        print(add_timestamp(color_text(f"Relevant classes: {list(all_relevant_classes)}", "blue")))
383
384        pruned_schema_data = yaml_schema.copy()
385        pruned_schema_data['classes'] = {}
386
387        # Populate pruned schema with only the relevant classes and their attributes
388        if 'classes' in yaml_schema:
389            for class_name, class_details in yaml_schema['classes'].items():
390                if class_name in all_relevant_classes:
391                    pruned_schema_data['classes'][class_name] = class_details.copy()
392                    # Ensure inherited classes are also included if they are not explicitly relevant
393                    # This is a simplification; a full LinkML pruning would trace dependencies
394                    is_a = class_details.get('is_a')
395                    if is_a and is_a not in pruned_schema_data['classes'] and is_a in yaml_schema['classes']:
396                        pruned_schema_data['classes'][is_a] = yaml_schema['classes'][is_a].copy()
397
398        if not pruned_schema_data['classes']:
399            print(add_timestamp(color_text("No classes left after pruning. Output YAML will be minimal.", "yellow")))
400
401        # Save the pruned schema to a new YAML file
402        if output_directory:
403            try:
404                create_directory_if_not_exists(output_directory)
405                output_file = os.path.join(output_directory, "linkml_schema_pruned.yaml")
406                with open(output_file, 'w', encoding='utf-8') as f:
407                    yaml.dump(pruned_schema_data, f, indent=2, sort_keys=False)
408                print(add_timestamp(color_text(f"Pruned LinkML schema saved to: {output_file}", "green")))
409
410                
411            except Exception as e:
412                print(add_timestamp(color_text(f"Error saving pruned schema to {output_file}: {e}", "red")))
413                return None
414        
415        return pruned_schema_data
